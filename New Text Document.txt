        // Royal Door Animation
        window.addEventListener('load', function() {
            setTimeout(() => {
                document.querySelector('.door-left').classList.add('open');
                document.querySelector('.door-right').classList.add('open');
                
                setTimeout(() => {
                    document.querySelector('.royal-doors').style.display = 'none';
                }, 2000);
            }, 500);

        // Enhanced Prismatic Light Rays with Flowing Curves
        const canvas = document.getElementById('prismaticCanvas');
        const ctx = canvas.getContext('2d');
        let mouseX = 0;
        let mouseY = 0;
        let rays = [];

        function resizeCanvas() {
            canvas.width = canvas.offsetWidth;
            canvas.height = canvas.offsetHeight;
            mouseX = canvas.width / 2;
            mouseY = canvas.height / 2;
            initRays();
        }

        function initRays() {
            rays = [];
            const numRays = 24;
            
            for (let i = 0; i < numRays; i++) {
                rays.push({
                    baseAngle: (Math.PI * 2 * i) / numRays,
                    color: `hsl(${(360 / numRays) * i}, 85%, 70%)`,
                    width: Math.random() * 3 + 2,
                    phase: Math.random() * Math.PI * 2,
                    speed: Math.random() * 0.02 + 0.01,
                    amplitude: Math.random() * 40 + 30,
                    length: Math.random() * 100 + 150,
                    segments: []
                });
            }
        }

        function drawRays() {
            ctx.clearRect(0, 0, canvas.width, canvas.height);
            
            const centerX = canvas.width / 2;
            const centerY = canvas.height / 2;
            
            const dx = mouseX - centerX;
            const dy = mouseY - centerY;
            const mouseDistance = Math.sqrt(dx * dx + dy * dy);
            const mouseAngle = Math.atan2(dy, dx);
            
            rays.forEach((ray, index) => {
                ray.phase += ray.speed;
                
                ctx.beginPath();
                ctx.moveTo(centerX, centerY);
                
                let angleDiff = mouseAngle - ray.baseAngle;
                while (angleDiff > Math.PI) angleDiff -= Math.PI * 2;
                while (angleDiff < -Math.PI) angleDiff += Math.PI * 2;
                
                const angleProximity = 1 - Math.abs(angleDiff) / Math.PI;
                const distanceInfluence = Math.min(mouseDistance / 200, 1);
                const totalInfluence = angleProximity * distanceInfluence * 0.6;
                
                const numPoints = 80;
                const points = [];
                
                for (let i = 0; i <= numPoints; i++) {
                    const t = i / numPoints;
                    
                    const currentAngle = ray.baseAngle + angleDiff * totalInfluence * (1 - Math.pow(1 - t, 2));
                    
                    const wave = Math.sin(ray.phase + t * Math.PI * 3) * ray.amplitude * (1 - t * 0.5);
                    const perpAngle = currentAngle + Math.PI / 2;
                    
                    const distance = ray.length * t;
                    let x = centerX + Math.cos(currentAngle) * distance;
                    let y = centerY + Math.sin(currentAngle) * distance;
                    
                    x += Math.cos(perpAngle) * wave * t;
                    y += Math.sin(perpAngle) * wave * t;
                    
                    if (mouseDistance < 300) {
                        const segmentDx = mouseX - x;
                        const segmentDy = mouseY - y;
                        const segmentDist = Math.sqrt(segmentDx * segmentDx + segmentDy * segmentDy);
                        
                        if (segmentDist < 200) {
                            const pull = (1 - segmentDist / 200) * 20 * t;
                            x += (segmentDx / segmentDist) * pull;
                            y += (segmentDy / segmentDist) * pull;
                        }
                    }
                    
                    points.push({ x, y, t });
                }
                
                if (points.length > 0) {
                    ctx.moveTo(points[0].x, points[0].y);
                    
                    for (let i = 1; i < points.length - 2; i++) {
                        const xc = (points[i].x + points[i + 1].x) / 2;
                        const yc = (points[i].y + points[i + 1].y) / 2;
                        ctx.quadraticCurveTo(points[i].x, points[i].y, xc, yc);
                    }
                    
                    if (points.length > 2) {
                        ctx.quadraticCurveTo(
                            points[points.length - 2].x,
                            points[points.length - 2].y,
                            points[points.length - 1].x,
                            points[points.length - 1].y
                        );
                    }
                }
                
                const gradient = ctx.createLinearGradient(
                    centerX, centerY,
                    points[points.length - 1].x,
                    points[points.length - 1].y
                );
                gradient.addColorStop(0, ray.color.replace('70%', '0%'));
                gradient.addColorStop(0.2, ray.color);
                gradient.addColorStop(0.8, ray.color);
                gradient.addColorStop(1, ray.color.replace('70%', '0%'));
                
                ctx.strokeStyle = gradient;
                ctx.lineWidth = ray.width;
                ctx.lineCap = 'round';
                ctx.lineJoin = 'round';

        function createSearchDropdown() {
            const searchContainer = document.querySelector('.search-container');
            const searchDropdown = document.createElement('div');
            searchDropdown.className = 'search-dropdown';
            searchDropdown.id = 'searchDropdown';
            searchContainer.appendChild(searchDropdown);
        }

        function performSearch(query) {
            const searchDropdown = document.getElementById('searchDropdown');
            
            if (!query || query.trim().length < 2) {
                searchDropdown.style.display = 'none';
                return;
            }
            
            const searchTerm = query.toLowerCase().trim();
            const results = searchData.pages.filter(page => {
                return page.title.toLowerCase().includes(searchTerm) ||
                       page.keywords.some(keyword => keyword.includes(searchTerm)) ||
                       page.content.toLowerCase().includes(searchTerm);
            });
            
            displaySearchResults(results, searchTerm);
        }

        function displaySearchResults(results, searchTerm) {
            const searchDropdown = document.getElementById('searchDropdown');
            
            if (results.length === 0) {
                searchDropdown.innerHTML = `
                    <div class="search-result-item no-results">
                        <i class="fas fa-search"></i>
                        <div>
                            <strong>No results found</strong>
                            <p>Try different keywords</p>
                        </div>
                    </div>
                `;
                searchDropdown.style.display = 'block';
                return;
            }
            
            let resultsHTML = `<div class="search-results-header">Found ${results.length} result${results.length > 1 ? 's' : ''}</div>`;
            
            results.forEach(result => {
                const highlightedTitle = highlightText(result.title, searchTerm);
                const highlightedContent = highlightText(result.content, searchTerm);
                
                resultsHTML += `
                    <a href="${result.url}" class="search-result-item">
                        <i class="fas fa-file-alt"></i>
                        <div class="search-result-content">
                            <strong>${highlightedTitle}</strong>
                            <p>${highlightedContent}</p>
                        </div>
                    </a>
                `;
            });
            
            searchDropdown.innerHTML = resultsHTML;
            searchDropdown.style.display = 'block';
        }

        function highlightText(text, searchTerm) {
            const regex = new RegExp(`(${searchTerm})`, 'gi');
            return text.replace(regex, '<mark>$1</mark>');
        }

        function createParticles() {
            const container = document.getElementById('heroParticles');
            for (let i = 0; i < 30; i++) {
                const particle = document.createElement('div');
                particle.className = 'particle';
                particle.style.left = Math.random() * 100 + '%';

                particle.style.height = particle.style.width;

            }
        }

        let currentSlide = 0;
        const slides = document.querySelectorAll('.hero-slide');
        const dots = document.querySelectorAll('.control-dot');
        let slideInterval;

        function showSlide(index) {
            slides.forEach((slide, i) => {
                slide.classList.toggle('active', i === index);
            });
            dots.forEach((dot, i) => {
                dot.classList.toggle('active', i === index);
            });
            currentSlide = index;
        }

        function nextSlide() {
            currentSlide = (currentSlide + 1) % slides.length;
            showSlide(currentSlide);
        }

        function changeSlide(direction) {
            currentSlide = (currentSlide + direction + slides.length) % slides.length;
            showSlide(currentSlide);
        }

        function goToSlide(index) {
            showSlide(index);
        }

        function startSlideshow() {
            slideInterval = setInterval(nextSlide, 6000);
        }

        function stopSlideshow() {
            clearInterval(slideInterval);
        }
